461 int CheckAvailable(std::vector<t *> res,double startingHour, double endHour,int day, int month)
462 {
463  for(size_t i =0; i < res.size(); i++)
464  {
465  if(res.at(i)->getDay() == day && res.at(i)->getMonth() == month)
466  {
467  //res- values from reservation already made
468  double resStart = res.at(i)->getStartingHour();
469  double resEnd = calculateEndHour(resStart,res.at(i)->getDuration());
470 
471  if(startingHour >= resStart && endHour <= resEnd)
472  // tempo da reserva esta em espaço ocupado
473  throw(InsideRes(resStart,endHour));
474  else if(startingHour <= resStart && endHour >= resStart)
475  //tempo da reserva entra em espaço ocupado
476  throw(EndHourInsideRes(resStart,resEnd));
477  else if(startingHour >= resStart && endHour >= resEnd && startingHour <= resEnd)
478  //tempo da reserva esta a meio
479  throw(StartHourInsideRes(startingHour, endHour));
480  }
481  }
482  return 0;
