427 template< class t>
428 int CheckAvailable(std::vector<t *> res,double startingHour, double endHour,int day, int month)
429 {
430  for(size_t i =0; i < res.size(); i++)
431  {
432  if(res.at(i)->getDay() == day && res.at(i)->getMonth() == month)
433  {
434  //res- values from reservation already made
435  double resStart = res.at(i)->getStartingHour();
436  double resEnd = calculateEndHour(resStart,res.at(i)->getDuration());
437 
438  if(startingHour >= resStart && endHour <= resEnd)
439  // tempo da reserva esta em espaço ocupado
440  throw(InsideRes(resStart,endHour));
441  else if(startingHour <= resStart && endHour >= resStart)
442  //tempo da reserva entra em espaço ocupado
443  throw(EndHourInsideRes(resStart,resEnd));
444  else if(startingHour >= resStart && endHour >= resEnd && startingHour <= resEnd)
445  //tempo da reserva esta a meio
446  throw(StartHourInsideRes(startingHour, endHour));
447  }
448  }
449  return 0;
450 }
